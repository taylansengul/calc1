\subsection*{Sample Exam 1}
\begin{enumerate}
  \item Find the following limits if they exist.
  \begin{enumerate}
    \item[a)]
    \[
      \lim_{x \to \infty} \left( \frac{x^2}{x+1} - \frac{x^2}{x-1} \right)
    \]
    \item[b)]
    \[
      \lim_{x  \to 0} \frac{x}{\abs{x-1} - \abs{x+1}}
    \]
    \item[c)]
    \[
      \lim_{x \to -\infty} \left( x + \sqrt{x^2-4x+1} \right)
    \]
  \end{enumerate}
  \item Show that $f(x)=x^3+x-1$ has a zero between $x=0$ and $x=1$.
  \item Find the slope of the tangent line to the curve
  \[
    \tan(x y^2) = \frac{2xy}{\pi}
  \]
  at the point $(-\pi, 1/2)$.
  \item Calculate the derivatives.
  \begin{enumerate}
    \item[a)]
    \[
      y=\frac{1+\sqrt{x}}{x^4+1}
    \]
    \item[b)]
    \[
      y=\left( \sin(\sqrt{x}) +1 \right)^3 + 7 x \cos x
    \]
  \end{enumerate}
  \item Is the function $y=\abs{x-2}$ differentiable at $x=2$? Show your work using the limit definition of the derivative.
\end{enumerate}
