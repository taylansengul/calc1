\subsection*{Sample Exam 2}
\begin{enumerate}
  \item Find the dimensions of the right triangle with hypotenuse $h=2$ and maximum area.
  \item Find the min and max values of $f(x) = 2x^3 - 15 x^2 + 24x +19$ on the interval $[0, 5]$
  \item Let $f(x) = x^4 - 3x^2 +2$.
  \begin{enumerate}
    \item Find the intervals on which $f$ is increasing or decreasing. Find all local extrema for $f$.
    \item Find the intervals on which $f$ is concave up or down. Find all inflection points for $f$.
    \item Sketch a graph of $f(x)$ using parts (a) and (b).
  \end{enumerate}
  \item Find all the horizontal and vertical asymptotes for
  \[
    y = \frac{x^2+3}{1-3x^2}.
  \]
  \item Let $f(x) = 2x^2 + x^3, x>0$. Show that $f$ is invertible and find $(f^{-1})'(16)$.
  \item Find $dy/dx$ by implicit differentiation if
  \[
    y^2 e^x + y \ln x = 2.
  \]
  \item Let $y = (1/x)^{\ln x}$. Find $dy/dx$.
  \item Find $\cos(\sin^{-1} 0.7)$.
\end{enumerate}
